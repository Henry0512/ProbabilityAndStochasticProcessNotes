\chapter{Stationary Stochastic Processes}

\section{Strict Stationary Processes}

\begin{definition}[Strict-Sense Stationary Process]\index{strict-sense stationary process}
  A \emph{strict-sense stationary process} (SSS) is a stochastic process \(\{X_t\}_{t \in T}\) such that for any constant \(h\), for any positive integer \(n\), for any \(t_1, t_2, \dots, t_n \in T\), and for any \(t_1 + h, t_2 + h, \dots, t_n + h \in T\),
  \[
    (X_{t_1}, X_{t_2}, \dots, X_{t_n})
  \]
  and
  \[
    (X_{t_1 + h}, X_{t_2 + h}, \dots, X_{t_n + h})
  \]
  have the same \(n\)th-order probability distribution function, i.e.,
  \[
    F(x_1, x_2, \dots, x_n; t_1, t_2, \dots, t_n) = F(x_1, x_2, \dots, x_n; t_1 + h, t_2 + h, \dots, t_n + h),
  \]
  for \(x_1, x_2, \dots, x_n \in \R\).
\end{definition}

\begin{theorem}
  Suppose that the process \(\{X_t\}_{t \in T}\) is strict-sense stationary and of second moments. Then,
  \begin{enumerate}
    \item for all \(t \in T\), \(\mu_X(t) = \E[X_t]\) is a constant.
    \item \(R_X(s, t) = \E[X_s X_t]\) depends only on the time difference \(t - s\) for all \(s, t \in T\).
  \end{enumerate}
\end{theorem}

Furthermore,
\begin{align*}
  C_X(t, t + \tau) &= \E[(X_t - \mu_X(t))(X_{t + \tau} - \mu_X(t + \tau))] \\
  &= \E[X_t X_{t + \tau}] - \E[X_t \mu_X] - \E[\mu_X X_{t + \tau}] + \E[\mu_X^2] \\
  &= R_X(\tau) - \mu_X^2 = C_X(\tau)
\end{align*}
depends only on \(\tau\) and
\[
  \sigma_X^2(t) = C_X(0) = R_X(0) - \mu_X^2
\]
is a constant for all \(t \in T\).

\begin{example}
  Supppose that \(\{X_n\}_{n \geq 0}\) are IID random sequences, each of which has a uniform distribution in the interval \((0, 1)\). Is \(\{X_n\}_{n \geq 0}\) a strict-sense stationary process? Determine the value \(\E[X_n]\) and \(\E[X_n X_m], n, m = 0, 1, 2, \dots\).
\end{example}
\begin{solution}
  Suppose that the distribution function of \(X_n\) is \(F(x)\). Then for all \(k, h \in \N\) and \(0 < n_1 < n_2 < \dots < n_k\), the joint distribution function of \((X_{n_1}, X_{n_2}, \dots, X_{n_k})\) and \((X_{n_1 + h}, X_{n_2 + h}, \dots, X_{n_k + h})\) are both
  \begin{align*}
    F(x_1, x_2, \dots, x_k) &= P(X_{n_1} \leq x_1, X_{n_2} \leq x_2, \dots, X_{n_k} \leq x_k) \\
    &= P(X_{n_1} \leq x_1) P(X_{n_2} \leq x_2) \cdots P(X_{n_k} \leq x_k) \\
    &= F(x_1) F(x_2) \cdots F(x_k).
  \end{align*}
  Thus, \(\{X_n\}_{n \geq 0}\) is a strict-sense stationary process.

  The expected value of \(X_n\) is
  \[
    \E[X_n] = \frac{1}{2},
  \]
  since \(X_n\) is uniformly distributed on the interval \((0, 1)\).

  For \(n = m\),
  \[
    \E[X_n X_m] = \E[X_n^2] = \int_0^1 x^2 \, \dd x = \frac{1}{3}.
  \]
  
  For \(n \neq m\),
  \[
    \E[X_n X_m] = \E[X_n] \E[X_m] = \left(\frac{1}{2}\right)^2 = \frac{1}{4},
  \]
  due to the independence of \(X_n\) and \(X_m\).
\end{solution}

\section{Wide Stationary Processes}

\begin{definition}
  A process \(\{X_t\}_{t \in T}\) is called a \emph{wide-sense stationary process} (WSS) if it is of second order and satisfies the following conditions:
  \begin{enumerate}
    \item for all \(t \in T\), \(\mu_X(t) = \E[X_t]\) is a constant.
    \item for all \(t, t + \tau \in T\), \(R_X(t, t + \tau) = R_X(\tau)\) depends only on the time difference \(\tau\).
  \end{enumerate}
  In particular, a random sequence \(\{X_n\}_{n \geq 0}\) is called a stationary random sequence (or time sequence) in the wide sense if it is of second order and satisfies the following conditions:
  \begin{enumerate}
    \item for all \(n \geq 0\), \(\E[X_n]\) is a constant.
    \item for all \(m, n \geq 0\), \(R_X(n, n + m) = R_X(m)\) depends only on the time difference \(m\).
  \end{enumerate}
\end{definition}

\begin{example}
  Suppose that \(\{X_n\}_{n \geq 1}\) are IID random sequences, each of which has a standard normal distribution, and \(\{Y_n\}_{n \geq 1}\) are IID random sequences, each of which has a uniform distribution in the interval \((-\sqrt{3}, \sqrt{3})\). Suppose that \(\{X_n\}_{n \geq 1}\) and \(\{Y_n\}_{n \geq 1}\) are independent. Let
  \[
    Z_n = \begin{cases}
      X_n, & \text{if } n \text{ is odd}, \\
      Y_n, & \text{if } n \text{ is even}.
    \end{cases}
  \]
  Prove that \(\{Z_n\}_{n \geq 1}\) is WSS but not SSS.
\end{example}
\begin{proof}
  First,
  \[
    \E[Z_n] = \begin{cases}
      \E[X_n] = 0, & \text{if } n \text{ is odd}, \\
      \E[Y_n] = 0, & \text{if } n \text{ is even},
    \end{cases}
  \]
  so the mean of \(Z_n\) is a constant.
  Next, for the autocorrelation function of \(Z_n\), for all \(n, m \geq 1\),
  \[
    R_Z(n, m) = \Cov(Z_n, Z_m) + \E[Z_n] = \Cov(Z_n, Z_m).
  \]
  \begin{enumerate}
    \item If \(n = m\), then
    \[
      R_Z(n, m) = \Var(Z_n) = \begin{cases}
        1, & \text{if } n \text{ is odd}, \\
        1, & \text{if } n \text{ is even},
      \end{cases}
    \]
    since \(Y_n \sim \mathrm{U}(-\sqrt{3}, \sqrt{3})\) has variance \(\frac{(2\sqrt{3})^2}{12} = 1\).
    \item If \(n \neq m\), then \(Z_n\) and \(Z_m\) are independent because they depend on different independent random sequences. Thus,
    \[
      R_Z(n, m) = \Cov(Z_n, Z_m) = 0.
    \]
  \end{enumerate}
  Therefore, for all \(n, m \geq 1\),
  \[
    R_Z(n, m) = \begin{cases}
      1, & n = m, \\
      0, & n \neq m,
    \end{cases}
  \]
  which depends only on the time difference \(|n - m|\). Hence, \(\{Z_n\}_{n \geq 1}\) is WSS.

  However, for any odd integer \(k \geq 1\),
  \[
    F(x; k) = P(Z_k \leq x) = P(X_k \leq x)
  \]
  and
  \[
    F(x; k + 1) = P(Z_{k + 1} \leq x) = P(Y_{k + 1} \leq x)
  \]
  are different distribution functions. Thus, the joint distribution functions of any finite collection of random variables in the process are not invariant under time shifts. Hence, the process is not SSS.
\end{proof}

\begin{proposition}
  Let \(X_t\) be a WSS process and relabel \(R_X(t, t + \tau) = R_X(\tau)\). Then \(R_X(\tau)\) has the following properties:
  \begin{enumerate}
    \item \(R_X(\tau)\) is even, i.e., \(R_X(-\tau) = R_X(\tau)\) for all \(\tau\).
    \item \(R_X(0) \geq 0\).
    \item \(|R_X(\tau)| \leq R_X(0)\) for all \(\tau\).
    \item If \(X_t\) is a periodic process with period \(T\), i.e., \(X_{t + T} = X_t\) for all \(t\), then its autocorrelation function is also periodic with period \(T\), i.e., \(R_X(\tau + T) = R_X(\tau)\) for all \(\tau\).
  \end{enumerate}
\end{proposition}

\begin{proposition}
  A function \(R(\tau)\) is the autocorrelation function of a WSS process if and only if it is an even function and is non-negative definite, i.e., for all \(n \in \N\), for all \(t_1 < t_2 < \dots < t_n\), and for all real vector \((a_1, a_2, \dots, a_n)\),
  \[
    \sum_{i = 1}^n \sum_{j = 1}^n a_i a_j R(t_i - t_j) \geq 0.
  \]
\end{proposition}

\begin{example}
  Suppose that \(\{X_n\}_{n \in \Z}\) is a pairwise uncorrelated random sequence, i.e., \(C_X(n, m) = 0\) for all \(n \neq m\), and \(\E[X_n] = 0\), \(\Var(X_n) = \sigma^2\) for all \(n\). Prove that \(\{X_n\}_{n \in \Z}\) is WSS.
\end{example}
\begin{proof}
  For all \(n \in \Z\),
  \[
    \E[X_n] = 0,
  \]
  and for all \(n, m \in \Z\),
  \[
    R_X(n, m) = \begin{cases}
      \sigma^2, & n = m, \\
      0, & n \neq m.
    \end{cases}
  \]
  
  Thus, for all \(n, m \in \Z\),
  \[
    R_X(n, m) = R_X(|n - m|).
  \]
  
  Therefore, \(\{X_n\}_{n \in \Z}\) is WSS.
\end{proof}

\begin{example}\label{ex:cosine_wave_random_phase}
  Suppose that \(\{X_t\}\) is a stationary process and
  \[
    Y_t = X_t \cos(\omega_0 t + \Theta),
  \]
  where \(X_t\) and \(\Theta\) are independent, \(\Theta \sim \mathrm{U}(0, 2\mpi)\), and \(\omega_0\) is a constant. Prove that \(\{Y_t\}\) is also a stationary process.
\end{example}
\begin{proof}
  For all \(t\),
  \[
    \E[Y_t] = \E[X_t] \E[\cos(\omega_0 t + \Theta)] = 0,
  \]
  since \(\Theta\) is uniformly distributed in the interval \((0, 2\mpi)\).

  For all \(s, t\),
  \begin{align*}
    R_Y(s, t) &= \E[Y_s Y_t] = \E[X_s X_t] \E[\cos(\omega_0 s + \Theta) \cos(\omega_0 t + \Theta)] \\
    &= R_X(s, t) \cdot \E\left[\frac{\cos(\omega_0 (s - t)) + \cos(\omega_0 (s + t) + 2\Theta)}{2}\right] \\
    &= R_X(s, t) \cdot \frac{\cos(\omega_0 (s - t))}{2},
  \end{align*}
  since the expected value of the second term involves integrating a cosine function over two periods, which results in zero:
  \[
    \E[\cos(\omega_0 (s + t) + 2\Theta)] = \int_0^{2\mpi} \cos(\omega_0 (s + t) + 2\theta) \cdot \frac{1}{2\mpi} \, \dd\theta = 0.
  \]
  Thus, \(R_Y(s, t)\) depends only on the time difference \(s - t\).

  Therefore, \(\{Y_t\}\) is a stationary process.
\end{proof}

\section{Power Spectral Density of Stationary Processes}

\begin{definition}[Fourier Transform]\index{Fourier transform}\index{frequency spectrum}
  Suppose that \(x_t (t \in \R)\) is a deterministic real-valued function satisfying the Dirichlet condition. Then the Fourier transform of \(x_t\) defined by
  \[
    F(\omega) = \int_{-\infty}^\infty x_t \me^{-\mi \omega t} \, \dd t.
  \]
  The Fourier transform \(F(\omega)\) of \(x_t\) is often called the \emph{frequency spectrum} of \(x_t\).
\end{definition}

\(F(\omega)\) is a complex-valued function of \(\omega\) and
\[
  F(-\omega) = \int_{-\infty}^\infty x_t \me^{\mi \omega t} \, \dd t = \overline{F(\omega)},
\]
where \(\overline{F(\omega)}\) is the complex conjugate of \(F(\omega)\). Thus, \(|F(-\omega)|^2 = |F(\omega)|^2\).

\begin{definition}[Inverse Fourier Transform]\index{inverse Fourier transform}
  Suppose that \(F(\omega)\) is the Fourier transform of a deterministic real-valued function \(x_t\). Then the inverse Fourier transform of \(F(\omega)\) defined by
  \[
    x_t = \frac{1}{2\mpi} \int_{-\infty}^\infty F(\omega) \me^{\mi \omega t} \, \dd\omega.
  \]
\end{definition}

\begin{table}
  \caption{Fourier transforms of some common functions}
  \begin{center}
    \begin{tabular}{c c}
      \toprule
      \(x_t\) & \(F(\omega)\) \\
      \midrule
      \(\delta(t)\) & \(1\) \\
      \(1\) & \(2\mpi \delta(\omega)\) \\
      \(\me^{-\alpha |t|}\) & \(\frac{2\alpha}{\omega^2 + \alpha^2}\) \\
      \(\me^{-\alpha t} u(t)\) & \(\frac{1}{\alpha + \mi \omega}\) \\
      \(\sin(\omega_0 t)\) & \(-\mi \pi [\delta(\omega - \omega_0) - \delta(\omega + \omega_0)]\) \\
      \(\cos(\omega_0 t)\) & \(\pi [\delta(\omega - \omega_0) + \delta(\omega + \omega_0)]\) \\
      \bottomrule
    \end{tabular}
  \end{center}
\end{table}

\begin{theorem}[Parseval's Theorem]\index{Parseval's theorem}
  Suppose that \(x(t)\) is a deterministic real-valued function with Fourier transform \(F(\omega)\). Then
  \[
    \int_{-\infty}^\infty x^2(t) \, \dd t = \frac{1}{2\mpi} \int_{-\infty}^\infty |F(\omega)|^2 \, \dd\omega.
  \]
\end{theorem}
\begin{remark}\index{energy spectral density}
  \(|F(\omega)|^2\) is often called the \emph{energy spectral density} of \(x(t)\).
\end{remark}

\begin{definition}[Continuity in Mean Square]\index{continuity in mean square}
  A process \(\{X_t\}_{t \in T}\) is said to be \emph{continuous in mean square} at \(t_0 \in T\) if
  \[
    \lim_{\varepsilon \to 0} \E\left[(X_{t_0 + \varepsilon} - X_{t_0})^2\right] = 0,
  \]
  denoted by
  \[
    \mathrm{l.i.m.}_{t \to t_0} X_t = X_{t_0},
  \]
  where \(\mathrm{l.i.m.}\) stands for the limit in mean square. The stochastic process \(\{X_t\}_{t \in T}\) is said to be \emph{mean square continuous process} if it is continuous in mean square at every \(t_0 \in T\).
\end{definition}

\begin{theorem}
  A stochastic process \(X_t\) is a mean square continuous process if and only if its autocorrelation function \(R_X(t_1, t_2)\) is continuous.
\end{theorem}

\begin{definition}[Mean Square Derivative]\index{mean square derivative}
  Suppose that \(X_t\) is a mean square continuous process. We call \(X_t\) has a \emph{mean square derivative} \(X_t'\) at \(t \in T\) if
  \[
    \mathrm{l.i.m.}_{\varepsilon \to 0} \frac{X_{t + \varepsilon} - X_t}{\varepsilon} = X_t',
  \]
  that is,
  \[
    \lim_{\varepsilon \to 0} \E\left[\left(\frac{X_{t + \varepsilon} - X_t}{\varepsilon} - X_t'\right)^2\right] = 0.
  \]
\end{definition}

\begin{theorem}
  Suppose that the mean and the autocorrelation function of \(X_t\) are \(\mu_X(t)\) and \(R_X(s, t)\), respectively. Then
  \[
    \mu_{X'}(t) = \E[X_t'] = \frac{\dd}{\dd t} \E[X_t] = \mu_X'(t),
  \]
  and
  \[
    R_{X'}(s, t) = \E[X_s' X_t'] = \frac{\partial^2}{\partial s \partial t} \E[X_s X_t] = \frac{\partial^2}{\partial s \partial t} R_X(s, t).
  \]
\end{theorem}

\begin{theorem}
  If \(X_t\) is a WSS process, then it is mean square continuous if and only if its autocorrelation function \(R_X(\tau)\) is continuous at \(\tau = 0\).
\end{theorem}

\begin{definition}[Average Power]\index{average power}
  Suppose that \(X_t\) is a mean-square continuous stationary process. The \emph{average power} of \(X_t\) is defined as
  \[
    Q := \lim_{T \to \infty} \E \left[\frac{1}{2T} \int_{-T}^T X_t^2 \, \dd t\right].
  \]
\end{definition}

\begin{definition}[Power Spectral Density]\index{power spectral density}
  Suppose that \(X_t\) is a mean-square continuous stationary process. The \emph{power spectral density} of \(X_t\) is defined as
  \[
    S_X(\omega) = \lim_{T \to \infty} \frac{1}{2T} \E\left[|F_T(\omega)|^2\right],
  \]
  where
  \[
    F_T(\omega) = \int_{-T}^T X_t \me^{-\mi \omega t} \, \dd t.
  \]
\end{definition}

\begin{proposition}
  Let \(X_t\) be a stationary process with autocorrelation function \(R_X(\tau)\), then the average power of \(X_t\) is
  \[
    Q = \lim_{T \to \infty} \E\left[\frac{1}{2T} \int_{-T}^T X_t^2 \, \dd t\right] = \lim_{T \to \infty} \frac{1}{2T} \int_{-T}^T \E[X_t^2] \, \dd t = R_X(0).
  \]
\end{proposition}

\begin{example}
  Find the average power for the sine wave with random phase
  \[
    X_t = a \sin(\omega_0 t + \Theta),
  \]
  where \(a\) and \(\omega_0\) are constants and \(\Theta\) is a random variable uniformly distributed in the interval \((0, 2\mpi)\).
\end{example}
\begin{proof}[Solution using the definition of average power]
  The average power of \(X_t\) is
  \[
    Q = \lim_{T \to \infty} \E\left[\frac{1}{2T} \int_{-T}^T a^2 \sin^2(\omega_0 t + \Theta) \, \dd t\right] = \frac{a^2}{2},
  \]
  since the expected value of \(\sin^2(\omega_0 t + \Theta)\) is
  \begin{align*}
    \E[\sin^2(\omega_0 t + \Theta)] &= \E\left[\frac{1 - \cos(2(\omega_0 t + \Theta))}{2}\right] \\
    &= \int_0^{2\mpi} \frac{1 - \cos(2(\omega_0 t + \theta))}{2} \cdot \frac{1}{2\mpi} \, \dd\theta = \frac{1}{2}. \qedhere
  \end{align*}
\end{proof}
\begin{proof}[Solution using the autocorrelation function]
  The mean of \(X_t\) is
  \[
    \mu_X(t) = \E[X_t] = \E[a \sin(\omega_0 t + \Theta)] = 0,
  \]
  since the integral of \(\sin(\omega_0 t + \theta)\) over one period is zero.

  Similar to the calculation in Example~\ref{ex:cosine_wave_random_phase}, the autocorrelation function of \(X_t\) is
  \[
    R_X(s, t) = \E[X_s X_t] = a^2 \E[\sin(\omega_0 s + \Theta) \sin(\omega_0 t + \Theta)] = \frac{a^2}{2} \cos(\omega_0 (s - t)),
  \]
  Thus, the autocorrelation function of \(X_t\),
  \[
    R_X(\tau) = \frac{a^2}{2} \cos(\omega_0 \tau)
  \]
  depends only on the time difference \(\tau\).

  Therefore, \(X_t\) is a stationary process and its average power is
  \[
    Q = R_X(0) = \frac{a^2}{2}. \qedhere
  \]
\end{proof}

\begin{proposition}
  Some properties of the power spectral density \(S_X(\omega)\) are as follows:
  \begin{enumerate}
    \item \(S_X(\omega)\) is non-negative for all \(\omega\).
    \item \(S_X(\omega)\) is a real-valued and even function of \(\omega\), i.e., \(\overline{S_X(\omega)} = S_X(-\omega) = S_X(\omega)\).
    \item If \(S_X(\omega)\) is a rational function, then it can be expressed as a ratio of two polynomials in \(\omega\),
    \[
      S_X(\omega) = S_0 \frac{\omega^{2n} + a_{2n-2}\omega^{2n-2} + \dots + a_0}{\omega^{2m} + b_{2m-2}\omega^{2m-2} + \dots + b_0},
    \]
    where \(a_{2n - i}, b_{2m - i} (i = 2, 4, \dots, 2n, j = 2, 4, \dots, 2m)\) are constants and \(S_0 > 0, m > n\), and the denominator polynomial has no real roots.
  \end{enumerate}
\end{proposition}

\begin{theorem}[Wiener-Khinchin Theorem]\index{Wiener-Khinchin theorem}\label{thm:Wiener-Khinchin}
  Suppose that \(R_X(\tau)\) is the autocorrelation function of a WSS process \(\{X_t\}\) and \(S_X(\omega)\) is its power spectral density. If
  \[
    \int_{-\infty}^\infty |R_X(\tau)| \, \dd\tau < \infty,
  \]
  then
  \[
    S_X(\omega) = \int_{-\infty}^\infty R_X(\tau) \me^{-\mi \omega \tau} \, \dd\tau,
  \]
  and
  \[
    R_X(\tau) = \frac{1}{2\mpi} \int_{-\infty}^\infty S_X(\omega) \me^{\mi \omega \tau} \, \dd\omega.
  \]
\end{theorem}

\begin{example}
  Find the power spectral density for the following exponential autocorrelation function:
  \[
    R_X(\tau) = A \me^{-\beta |\tau|}
  \]
  with parameters \(A > 0\) and \(\beta > 0\).
\end{example}
\begin{solution}
  The power spectral density of \(X_t\) is
  \begin{align*}
    S_X(\omega) &= \int_{-\infty}^\infty A \me^{-\beta |\tau|} \me^{-\mi \omega \tau} \, \dd\tau = A \int_{-\infty}^0 \me^{\beta \tau} \me^{-\mi \omega \tau} \, \dd\tau + A \int_0^\infty \me^{-\beta \tau} \me^{-\mi \omega \tau} \, \dd\tau \\
    &= A\left[\frac{1}{\beta - \mi \omega} + \frac{1}{\beta + \mi \omega}\right] = A\frac{2\beta}{\beta^2 + \omega^2}. \qedhere
  \end{align*}
\end{solution}

\begin{example}
  Assume that the power spectral density of a WSS process \(X_t\) is
  \[
    S_X(\omega) = \frac{\omega^2 + 4}{\omega^4 + 10\omega^2 + 9}.
  \]
  Find the autocorrelation function of \(X_t\).
\end{example}
\begin{solution}
  The autocorrelation function of \(X_t\) is
  \begin{align*}
    R_X(\tau) &= \frac{1}{2\mpi} \int_{-\infty}^\infty \frac{\omega^2 + 4}{\omega^4 + 10\omega^2 + 9} \me^{\mi \omega \tau} \, \dd\omega = \frac{1}{2\mpi} \int_{-\infty}^\infty \frac{\omega^2 + 4}{(\omega^2 + 1)(\omega^2 + 9)} \me^{\mi \omega \tau} \, \dd\omega \\
    &= \frac{1}{2\mpi} \int_{-\infty}^\infty \left[\frac{3/8}{\omega^2 + 1} + \frac{5/8}{\omega^2 + 9}\right] \me^{\mi \omega \tau} \, \dd\omega = \frac{3}{16} e^{-|\tau|} + \frac{5}{48} e^{-3|\tau|}. \qedhere
  \end{align*}
\end{solution}

\begin{example}
  Find the power spectral density for the stochastic process \(X_t = a \cos(\omega_0 t + \Theta)\), where \(a\) and \(\omega_0\) are constants and \(\Theta\) is a random variable uniformly distributed in the interval \((0, 2\mpi)\).
\end{example}
\begin{solution}
  The mean of \(X_t\) is
  \[
    \mu_X(t) = \E[X_t] = \E[a \cos(\omega_0 t + \Theta)] = \int_0^{2\mpi} a \cos(\omega_0 t + \theta) \cdot \frac{1}{2\mpi} \, \dd\theta = 0,
  \]
  since the integral of \(\cos(\omega_0 t + \theta)\) over one period is zero.

  As calculated in the proof of Example~\ref{ex:cosine_wave_random_phase}, the autocorrelation function of \(X_t\) is
  \[
    R_X(s, t) = \E[X_s X_t] = a^2 \E[\cos(\omega_0 s + \Theta) \cos(\omega_0 t + \Theta)] = \frac{a^2}{2} \cos(\omega_0 (s - t)),
  \]
  Thus, the autocorrelation function of \(X_t\),
  \[
    R_X(\tau) = \frac{a^2}{2} \cos(\omega_0 \tau)
  \]
  depends only on the time difference \(\tau\).
  
  Thus, \(X_t\) is a WSS process. By Theorem~\ref{thm:Wiener-Khinchin}, the power spectral density of \(X_t\) is
  \begin{align*}
    S_X(\omega) &= \int_{-\infty}^\infty R_X(\tau) \me^{-\mi \omega \tau} \, \dd\tau = \frac{a^2}{2} \int_{-\infty}^\infty \cos(\omega_0 \tau) \me^{-\mi \omega \tau} \, \dd\tau \\
    &= \frac{\mpi a^2}{2} [\delta(\omega - \omega_0) + \delta(\omega + \omega_0)],
  \end{align*}
  where \(\delta(\cdot)\) is the Dirac delta function.
\end{solution}

\section{Linear and Time-Invariant Systems}

\begin{theorem}
  Let the stationary process \(\{X_t\}\) with mean \(\mu_X\) and autocorrelation function \(R_X(\tau)\) be the input to the filter with output \(\{Y_t\}\). The power spectral density of \(\{X_t\}\) is \(S_X(\omega)\) and the system's frequency response is \(H(\omega)\). Then the power spectral density of \(\{Y_t\}\) is
  \[
    S_Y(\omega) = |H(\omega)|^2 S_X(\omega).
  \]
  The cross-power spectral density of \(\{X_t\}\) and \(\{Y_t\}\) is
  \[
    S_{XY}(\omega) = H(\omega) S_X(\omega).
  \]
\end{theorem}
